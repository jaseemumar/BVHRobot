527  // Gauss-Jordan elimination with partial pivoting
528  mat33 mat33::inverse(void)
529  {
530  // As a evolves from original mat into identity
531  // b evolves from identity into inverse(a)
532  mat33 a(*this), b(identity2D());
533  int i, j, i1;
534 
535  // Loop over cols of a from left to right, eliminating above and below diag
536  for (j=0; j<3; j++)
537  {
538  // Find largest pivot in column j among rows j..2
539  i1 = j;
540  // Row with largest pivot candidate
541  for (i=j+1; i<3; i++)
542  if (fabs(a.row[i].data[j]) > fabs(a.row[i1].data[j]))
543  i1 = i;
544 
545  // Swap rows i1 and j in a and b to put pivot on diagonal
546  swap(a.row[i1], a.row[j]);
547  swap(b.row[i1], b.row[j]);
548 
549  // Scale row j to have a unit diagonal
550  if (fabs(a.row[j].data[j]) <= EPSILON)
551  throw util::common::fatal_error("mat33::inverse : Singular matrix, cannot invert");
552  b.row[j] /= a.row[j].data[j];
553  a.row[j] /= a.row[j].data[j];
554 
555  // Eliminate off-diagonal elems in col j of a, doing identical ops to b
556  for (i=0; i<3; i++)
557  if (i != j)
558  {
559  b.row[i] -= a.row[i].data[j] * b.row[j];
560  a.row[i] -= a.row[i].data[j] * a.row[j];
561  }
562  }
563  return b;
564  }
