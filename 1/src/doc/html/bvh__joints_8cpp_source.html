149 {
150  if (token_map_array::get_token(strtoken) == _ROOT)

152  joint_type = _root;
153  }
154  else if (token_map_array::get_token(strtoken) == _JOINT)
155  {
156  joint_type = _joint;
157  }
158  else if (strtoken == "End")
159  {
160  std::string temptoken;

162  if (token_map_array::get_token(strtoken+" "+temptoken) == _ENDSITE)

164  joint_type = _endsite;
165  }
166  }
167  else
168  throw new util::common::warning_error("(bvh::read) : Missing 'ROOT/JOINT/End Site' tag in bvh file");
169 
170  if (joint_type != _endsite) inp>>name;
171 
172  inp>>strtoken;
173 
174  if (token_map_array::get_token(strtoken) != _CURLY_BRACE_OPEN)
175  throw new util::common::warning_error("(bvh::read) : Missing '{' in bvh file");
176 
177  inp>>strtoken;
178 
179  if (token_map_array::get_token(strtoken) != _OFFSET)
180  throw new util::common::warning_error("(bvh::read) : Missing 'OFFSET' tag in bvh file");
181 
182  inp>>offset[util::math::X]>>offset[util::math::Y]>>offset[util::math::Z];
183 
184  if (joint_type == _root) absolute_offset = offset;
185 
186  if (joint_type != _endsite)
187  {
188  inp>>strtoken;
189  if (token_map_array::get_token(strtoken) != _CHANNELS)
190  throw new util::common::warning_error("(bvh::read) : Missing 'CHANNELS' tag in bvh file");
191 
192  inp>>channels.num_channels;

194  for(int i = 0; i < channels.num_channels; i++)

196  inp>>strtoken;

201  case _ZPOS: channels.ch_order[i] = _zpos; break;
202  case _XROT: channels.ch_order[i] = _xrot; break;
203  case _YROT: channels.ch_order[i] = _yrot; break;

207  }
208  }
209  }
